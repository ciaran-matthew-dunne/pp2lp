# -- Reflexivity (AR11/EVR4 + VR3 chain) --
ar_refl                a <= a
ar_refl_zero           0 <= 0
ar_refl_sub            1 - 1 <= 0
ar_eq_refl             a = a
ar_zero_eq             0 = 0
ar_one_eq              1 = 1
ar_add_zero_r          a + 0 = a
ar_add_zero_l          0 + a = a

# -- AR1: discharge a trivially-true (E ≤ E) antecedent. --
ar1_refl_disc          (a <= a) => b = b

# -- AR2: a > b ⇒ ¬(a ≤ b). PP encodes `>` as `not(_ <= _)`. --
ar2_neg_refl           not(a <= a) => false
ar2_double_neg_refl    not(not(a <= a))

# -- AR3: ¬(a ≤ 0) ⇒ 1 − a ≤ 0. Triggered via excluded-middle.
# (Compound antecedents like `a + b` expose a pre-existing AR3 emitter
# bug — a single-variable goal is the only reliable trigger.) --
ar3_em                 a <= 0 or not(a <= 0)

# -- AR4: F ≤ 0 in H, E + F > 0 ⇒ ¬(E ≤ 0). Discreteness. --
ar4_succ_above         (a <= 0 and not(a + 1 <= 1)) => false
ar4_pred_below         (not(a <= 0) and a + 1 - 1 <= 0) => false
ar4_chain              (a <= 0 and -1 <= 0 and not(a - 1 <= 0)) => false

# -- AR5/AR6: a ≤ 0 ∧ −a ≤ 0 ⇒ a = 0. Negation antisymmetry.
# (Goals concluding `a = 0` cause PP to chain AR12 onto an `=` antecedent —
# pre-existing emitter limitation. The reordered `0 = a` form sidesteps it.)
# AR5 vs AR6 differ only in which conjunct is hyp vs antecedent. --
ar5_neg_antisym_zero   (a <= 0 and -a <= 0) => 0 = a
ar6_neg_antisym_zero   (-a <= 0 and a <= 0) => 0 = a

# -- AR8: a − b ≤ 0; b − a ≤ 0 ⇒ a = b. Subtraction antisymmetry.
# (AR7's emitter currently fills `c` with the placeholder `𝟎` —
# AR7-only goals fail until that is fixed; AR8 is independent.) --
ar8_sub_antisym        (a - b <= 0 and b - a <= 0) => b = a
ar8_leq_antisym        (a <= b and b <= a) => b = a

# -- AR9 / AR10 (solver simplification, exercised throughout) --
ar_simp_zero           not(a <= 0 - 0) => not(a <= 0)
ar_simp_add_zero       not(a + 0 <= 0) => not(a <= 0)
ar_simp_zero_add       not(0 + a <= 0) => not(a <= 0)
