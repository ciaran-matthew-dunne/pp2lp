\providecommand{\titlerunning}[1]{}
\providecommand{\authorrunning}[1]{}

% ── System / tool / format names ──────────────────────────────────────
\newcommand{\pp}{PP}                       % the Predicate Prover
\newcommand{\ab}{Atelier~B}
\newcommand{\lp}{LambdaPi}                 % the proof assistant
\newcommand{\dedukti}{Dedukti}
\newcommand{\tool}{\texttt{pp2lp}}         % our tool
\newcommand{\bmethod}{B-Method}
\newcommand{\bbook}{B\nobreakdash-Book}
\newcommand{\lpcmr}{\ensuremath{\lambda\Pi}-calculus modulo rewriting}
\newcommand{\lpc}{\ensuremath{\lambda\Pi}}

% File-format names.
\newcommand{\trace}{\texttt{.trace}}
\newcommand{\replay}{\texttt{.replay}}
\newcommand{\butf}{\texttt{.but}}
\newcommand{\lpf}{\texttt{.lp}}

% Rule names, e.g. \rn{AND2}, \rn{ALL7}.
\newcommand{\rn}[1]{\textsf{#1}}

% Inline LP keywords / modifiers, matching the listings colours (preamble).
\newcommand{\lpkw}[1]{\textcolor{lpKeyword}{\bfseries\texttt{#1}}}
% Modifiers (constant/injective/opaque/sequential) keep their own colour but
% match the keyword weight (bold), so styling is uniform in weight.
\newcommand{\lpmod}[1]{\textcolor{lpModifier}{\bfseries\texttt{#1}}}
% Keyword / modifier terminals inside a math-mode grammar (\autoref{app:syntax}).
\newcommand{\kw}[1]{\text{\lpkw{#1}}}
\newcommand{\md}[1]{\text{\lpmod{#1}}}

% ── Logic / B-Book primitives (math mode) ─────────────────────────────
\newcommand{\TYPE}{\mathsf{TYPE}}
\newcommand{\KIND}{\mathsf{KIND}}
\newcommand{\Prf}{\pi}                     % proof-of:  π P
\newcommand{\El}{\tau}                     % type coercion:  τ ι
\newcommand{\Prop}{\mathsf{Prop}}
\newcommand{\Setu}{\mathsf{Set}}
\newcommand{\Res}{\mathsf{Res}}
\newcommand{\restm}{\mathsf{res\_tm}}
\newcommand{\Tuple}{\mathsf{Tuple}}
\newcommand{\prj}{\mathsf{prj}}
\newcommand{\trust}{\texttt{trust}}
\newcommand{\refine}{\texttt{refine}}
% ⋀ for prose/identifier use. JetBrains Mono lacks U+22C0, and the listings
% literate map only applies inside code boxes, so a raw ⋀ in a body \texttt{}
% renders as tofu — use \bwand (e.g. \texttt{\bwand\_init}) there instead.
\newcommand{\bwand}{\ensuremath{\bigwedge}}

% n-ary quantifier glyphs (matching the LP encoding).
\newcommand{\Dia}{\diamondsuit}            % ♢  n-ary ∀, goal side  (forall)
\newcommand{\Hrt}{\heartsuit}              % ♡  n-ary ∀, hyp side   (forall2)
\newcommand{\allI}{\mathop{!\!!}}          % !! single-var ∀ over Tuple 1
\newcommand{\xstI}{\mathop{?\!?}}          % ?? existential

% Sequent / yields.
\newcommand{\seq}{\vdash}
\newcommand{\yields}{\rightsquigarrow}     % ⇝  "yields result"
\newcommand{\app}{\mathbin{+\mkern-7mu+}}  % ++  list concatenation
\newcommand{\subst}[2]{[#1 \mathbin{:=} #2]}
