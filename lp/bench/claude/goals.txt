# claude — a handwritten pipeline-stress / coverage suite.
#
# This suite is for *testing the pp2lp pipeline*, not for staying green.  It
# deliberately spans the whole rule space (every PP family) and the whole
# proof-size span (one-rule leaves → 200+-rule proofs), AND it probes the known
# frontier: set-equality / equality-congruence, arithmetic antisymmetry and
# double-negation (AR4/AR7/AR8), INS over rewritten hyps, NAT membership, and
# existential/quantifier shapes REPLAY mishandles.  Goals expected to fail or be
# dropped are KEPT and marked — a failure here is a signal about the pipeline,
# not a defect in the suite.  Authored by hand, grouped by what each section
# tests.
#
# Format (shared with synth/nrm_test): name | kind | goal
#   prop  — bare identifiers are propositional variables
#   expr  — bare identifiers are expression variables (sets / functions / ints)
#
# Each goal is proved from itself + inferred `_delta_{p,e}` hypotheses over its
# free identifiers; promoting the goal-as-hypothesis drives PP normalisation.
# Note: a self-proof lets PP discharge a propositional tautology in one AXM8
# step (the promoted hypothesis matches the goal) — those are the trivial end of
# the size span; the real multi-rule work is in the expr / quantifier / set /
# arithmetic / contradiction goals, where hyp- and goal-normalisation diverge.
#
# `goals.txt` is the source of truth — regenerate with `pp2lp gen claude`,
# check with `pp2lp run claude`.  Coverage notes (→ RULE) are intended targets;
# measure actual firings from the generated `.replay` files.  Sections L–P below
# are the failure-frontier probes (✗/⚠ expected).

# ════════════════════════════════════════════════════════════════════════
# A. Propositional connectives — AND / OR / NOT, tiny → medium  (AND*, OR*, NOT*)
# ════════════════════════════════════════════════════════════════════════

# leaves (1–3 rules)
p_atom            | prop | a
p_and             | prop | a & b
p_or              | prop | a or b
p_not             | prop | not(a)
p_imp             | prop | a => b
p_iff             | prop | a <=> b

# identity / idempotence
p_imp_self        | prop | a => a
p_iff_self        | prop | a <=> a
p_and_idem        | prop | a & a <=> a
p_or_idem         | prop | a or a <=> a

# commutativity
p_and_comm        | prop | a & b <=> b & a
p_or_comm         | prop | a or b <=> b or a
p_iff_comm        | prop | (a <=> b) <=> (b <=> a)

# associativity
p_and_assoc       | prop | (a & b) & c <=> a & (b & c)
p_or_assoc        | prop | (a or b) or c <=> a or (b or c)

# distributivity (both directions)
p_distrib_ao      | prop | a & (b or c) <=> (a & b) or (a & c)
p_distrib_oa      | prop | a or (b & c) <=> (a or b) & (a or c)

# absorption
p_absorb_and      | prop | a & (a or b) <=> a
p_absorb_or       | prop | a or (a & b) <=> a

# De Morgan (2- and 3-ary)
p_demorgan_and    | prop | not(a & b) <=> not(a) or not(b)
p_demorgan_or     | prop | not(a or b) <=> not(a) & not(b)
p_demorgan_and3   | prop | not(a & b & c) <=> not(a) or not(b) or not(c)
p_demorgan_or3    | prop | not(a or b or c) <=> not(a) & not(b) & not(c)

# negation laws
p_dneg_elim       | prop | not(not(a)) => a
p_dneg_intro      | prop | a => not(not(a))
p_tneg            | prop | not(not(not(a))) => not(a)
p_excluded_middle | prop | a or not(a)
p_noncontra       | prop | not(a & not(a))

# bigger conjunction / disjunction shapes
p_conj4           | prop | a & b & c & d
p_disj4           | prop | a or b or c or d
p_conj_in_disj    | prop | (a & b) or c
p_disj_in_conj    | prop | (a or b) & c

# B. Implication structure — IMP family, medium → large
# ════════════════════════════════════════════════════════════════════════

imp_mp            | prop | a & (a => b) => b
imp_mt            | prop | (a => b) & not(b) => not(a)
imp_contrapos     | prop | (a => b) <=> (not(b) => not(a))
imp_curry         | prop | (a & b => c) <=> (a => b => c)
imp_permute       | prop | (a => b => c) => (b => a => c)
imp_to_or         | prop | (a => b) => not(a) or b
imp_from_or       | prop | not(a) or b => (a => b)
imp_neg           | prop | not(a => b) => a & not(b)
imp_peirce        | prop | ((a => b) => a) => a
imp_dist          | prop | (a => b => c) => ((a => b) => (a => c))
imp_cases         | prop | (a => c) & (b => c) => (a or b => c)
imp_syll          | prop | (a => b) & (b => c) => (a => c)
imp_syll4         | prop | (a => b) & (b => c) & (c => d) => (a => d)
imp_syll5         | prop | (a => b) & (b => c) & (c => d) & (d => e) => (a => e)
imp_chain_use     | prop | a & (a => b) & (b => c) & (c => d) => d
