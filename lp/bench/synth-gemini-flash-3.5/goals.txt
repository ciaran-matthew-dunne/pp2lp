# ---- Propositional and Implication Chains ----
gemini_demorgan_and      | prop | not(a & b) <=> (not(a) or not(b))
gemini_demorgan_or       | prop | not(a or b) <=> (not(a) & not(b))
gemini_contrapositive    | prop | (a => b) <=> (not(b) => not(a))
gemini_peirce            | prop | ((a => b) => a) => a
gemini_distrib_and_or    | prop | (a or (b & c)) <=> ((a or b) & (a or c))
gemini_implication_trans | prop | (a => b) & (b => c) => (a => c)

# ---- Set Theory & Distributivity ----
gemini_subset_singleton  | expr | {x} <: s & {y} <: s => {x,y} <: s

# ---- Equality & Functional Application ----
gemini_app_equality      | expr | f(g(x)) = y & g(x) = z => f(z) = y
gemini_nested_app_eq     | expr | f(x) = g(y) & g(y) = h(z) => f(x) = h(z)

# ---- Functions & Domain/Range ----
gemini_comp_domain       | expr | !x.(x: s => f(x): t) & !y.(y: t => g(y): u) => !x.(x: s => g(f(x)): u)
gemini_dom_defined       | expr | !x.(x: dom(f) => f(x): t)

# ---- Arithmetic Relations ----
gemini_add_commutative   | expr | x + y <= z => y + x <= z
gemini_neg_neg_bounds    | expr | x <= 0 & y <= 0 => x + y <= 0
gemini_leq_transitivity  | expr | x <= y & y <= z => x <= z
gemini_sub_equivalence   | expr | x - y <= 0 => x <= y

# ---- Complex Quantified Contradictions ----
gemini_double_contra     | expr | !x.(x: s => not(x: t)) & a: s & a: t => bfalse
gemini_vacuous_imply     | expr | !x.(x: s => !y.(y: t => a: u))
